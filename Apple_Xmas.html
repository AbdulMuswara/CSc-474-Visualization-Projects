import pandas as pd

from plotly.subplots import make_subplots

df = pd.read_csv("dashboard_data.csv")

# Style
BLUE = "#1f77ff"
LIGHT_BLUE = "#4da3ff"
GRAY = "#bfbfbf"
DARK_GRAY = "#666"

fig = make_subplots(
    rows=3, cols=3,
    specs=[
        [{"type": "domain"}, {"type": "domain"}, {"type": "xy"}],
        [{"type": "xy"}, {"type": "xy"}, {"type": "xy"}],
        [{"type": "xy"}, {"colspan": 2}, None]
    ],
    horizontal_spacing=0.06,
    vertical_spacing=0.12,
)

# Customer
seg = df[df.category == "customer_segment"]
fig.add_trace(
    go.Pie(
        labels=seg.label,
        values=seg.value,
        hole=0.55,
        textinfo="percent",
        marker=dict(colors=[BLUE, GRAY, "#d9d9d9"]),
        showlegend=False
    ),
    row=1, col=1
)

# Customer Gender
gen = df[df.category == "gender"]
fig.add_trace(
    go.Pie(
        labels=gen.label,
        values=gen.value,
        hole=0.55,
        textinfo="percent",
        marker=dict(colors=[BLUE, GRAY]),
        showlegend=False
    ),
    row=1, col=2
)

# Country
country = df[df.category == "country"]

fig.add_trace(
    go.Bar(
        x=country.value,
        y=country.label,
        orientation="h",
        marker=dict(color=BLUE),
        text=[f"€{v}M" for v in country.value],
        textposition="outside"
    ),
    row=1, col=3
)

# Covid Metrics
store = df[(df.category == "covid") & (df.type == "in_store")]
online = df[(df.category == "covid") & (df.type == "online")]

fig.add_trace(
    go.Bar(x=store.label, y=store.value, name="In-store", marker_color=BLUE),
    row=2, col=1
)

fig.add_trace(
    go.Bar(x=online.label, y=online.value, name="Online", marker_color=GRAY),
    row=2, col=1
)

# Shopping Method
shop = df[df.category == "shopping"]

colors = {"instore": BLUE, "online": "#9ec9ff", "other": "#d9d9d9"}

for t in ["instore", "online", "other"]:
    temp = shop[shop.type == t]
    fig.add_trace(
        go.Bar(
            y=temp.label,
            x=temp.value,
            orientation="h",
            marker_color=colors[t],
            name=t,
            text=[f"{v}%" for v in temp.value],
            textposition="inside"
        ),
        row=2, col=2
    )

# Payment Method
pay = df[df.category == "payment"]
