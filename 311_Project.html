df.columns = df.columns.str.lower().str.replace(" ", "_")

df = (
    df.assign(
        created_date=pd.to_datetime(df["created_date"], errors="coerce")
    )
    .dropna(subset=["created_date"])
    .assign(
        hour=lambda x: x["created_date"].dt.hour,
        day_of_week=lambda x: x["created_date"].dt.dayofweek,
        year=lambda x: x["created_date"].dt.year,
        borough=lambda x: x["borough"].str.upper().str.strip()
    )
)

BOROUGH_ORDER = ["MANHATTAN", "BROOKLYN", "QUEENS", "BRONX", "STATEN ISLAND"]

df = df[df["borough"].isin(BOROUGH_ORDER)]

import plotly.express as px

DAY_LABELS = ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"]

heatmap_data = (
    df.groupby(["day_of_week", "hour"])
      .size()
      .unstack(fill_value=0)
      .reindex(columns=range(24), fill_value=0)
)

fig_heatmap = px.imshow(
    heatmap_data,
    labels=dict(x="Hour of Day", y="Day of Week", color="Number of complaints"),
    x=[f"{h%12 or 12}{'am' if h < 12 else 'pm'}" for h in range(24)],
    y=DAY_LABELS,
    title="Noise Complaints by Day and Hour",
    color_continuous_scale="YlOrRd" # Changed color scale to Viridis
)

fig_heatmap.show()

borough_counts = (
    df["borough"]
    .value_counts()
    .reindex(BOROUGH_ORDER)
    .fillna(0)
)

PALETTE = {
    "MANHATTAN":     "#E63946",
    "BROOKLYN":      "#457B9D",
    "QUEENS":        "#2A9D8F",
    "BRONX":         "#E9C46A",
    "STATEN ISLAND": "#9B5DE5",
}

fig_bar = px.bar(
    x=[b.title() for b in BOROUGH_ORDER],
    y=borough_counts,
    color=BOROUGH_ORDER,
    color_discrete_map=PALETTE,
    title="Noise Complaints by Borough (2020 – Present)",
    labels={"x": "Borough", "y": "Total Noise Complaints"}
)

fig_bar.show()

current_year = pd.Timestamp.now().year

yearly = (
    df.query("year < @current_year")
      .groupby(["year", "borough"])
      .size()
      .reset_index(name="count")
)

fig_line = px.line(
    yearly,
    x="year",
    y="count",
    color="borough",
    color_discrete_map=PALETTE,
    markers=True,
    title="Noise Complaints Over Time by Borough",
    labels={"year": "Year", "count": "Number of Complaints"}
)

fig_line.show()
