# Make sure runtime is numeric
df['Runtime'] = pd.to_numeric(df['Runtime_Minutes'], errors='coerce')

# Aggregate total watch time per month
watch_time = df.groupby('Month')['Runtime'].sum().reset_index()

fig1 = px.line(
    watch_time,
    x='Month',
    y='Runtime',
    markers=True,
    color_discrete_sequence=['#007bff'],
    title="Total Watch Time Per Month"
)

fig1.update_layout(
    title_x=0.5,
    xaxis_title="Month",
    yaxis_title="Total Watch Time (Minutes)",
    template="plotly_white"
)

fig1.show()

genre_counts = df['Genre'].value_counts().reset_index()
genre_counts.columns = ['Genre', 'Count']

fig2 = px.bar(
    genre_counts.sort_values('Count'),
    x='Count',
    y='Genre',
    orientation='h',
    color='Genre',
    title="Most Watched Movie Genres",
    text='Count'
)

fig2.update_traces(textposition='outside')

fig2.update_layout(
    title_x=0.5,
    xaxis_title="Number of Movies",
    yaxis_title="Genre",
    showlegend=False,
    template="plotly_white"
)

fig2.show()
